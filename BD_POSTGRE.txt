-- Tabla de administradores
CREATE TABLE administrador (
    idAdministrador SERIAL PRIMARY KEY, 
    name VARCHAR(100),
    mail VARCHAR(100),
    clave VARCHAR(60)
);

-- Tabla de profesores
CREATE TABLE profesor (
    idProfesor SERIAL PRIMARY KEY, 
    name VARCHAR(100),
	phone INTEGER,
    mail VARCHAR(100),
    clave VARCHAR(60)
);

-- Tabla de estudiantes
CREATE TABLE estudiante (
    idEstudiante SERIAL PRIMARY KEY,
    name VARCHAR(100),
    dni VARCHAR(8),
    address VARCHAR(100),
    mail VARCHAR(100),
    clave VARCHAR(60)
);

-- Tabla de cursos
CREATE TABLE curso (
    idCurso SERIAL PRIMARY KEY,
    idProfesor INT,
    nameCurso VARCHAR(100),
    codigo VARCHAR(100),
    descripcion TEXT
);
